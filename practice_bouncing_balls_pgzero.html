# List to store all the balls

def draw():

def update():
    for i, ball in enumerate(balls):
        # Update ball position
        ball["x"] += ball["vx"]
        ball["y"] += ball["vy"]

        # Check for wall collisions
        if ball["x"] - ball["radius"] < 0 or ball["x"] + ball["radius"] > WIDTH:
            ball["vx"] = -ball["vx"]
        if ball["y"] - ball["radius"] < 0 or ball["y"] + ball["radius"] > HEIGHT:
            ball["vy"] = -ball["vy"]

        # Check for collisions with other balls
        for j, other in enumerate(balls):
            if i != j:
                dx = ball["x"] - other["x"]
                dy = ball["y"] - other["y"]
                distance = math.sqrt(dx**2 + dy**2)
                if distance < ball["radius"] + other["radius"]:
                    # Handle elastic collision
                    nx, ny = dx / distance, dy / distance
                    p = ball["vx"] * nx + ball["vy"] * ny - other["vx"] * nx - other["vy"] * ny
                    ball["vx"] -= p * nx
                    ball["vy"] -= p * ny
                    other["vx"] += p * nx
                    other["vy"] += p * ny

def on_mouse_down(pos):
    global dragged_ball
    # Create a new ball at the mouse position
    new_ball = {
        "x": pos[0],
        "y": pos[1],
        "vx": 0,
        "vy": 0,
        "radius": random.randint(10, 30),
        "color": (random.randint(100, 255), random.randint(100, 255), random.randint(100, 255)),
    }
    balls.append(new_ball)
    dragged_ball = new_ball

def on_mouse_move(pos):
    global dragged_ball
    if dragged_ball:
        # Set the velocity based on the drag distance
        dragged_ball["vx"] = (pos[0] - dragged_ball["x"]) / 10
        dragged_ball["vy"] = (pos[1] - dragged_ball["y"]) / 10

def on_mouse_up():
    global dragged_ball
    dragged_ball = None

def update():
    for i, ball in enumerate(balls):
        # Update ball position
        ball["x"] += ball["vx"]
        ball["y"] += ball["vy"]

        # Check for wall collisions
        if ball["x"] - ball["radius"] < 0 or ball["x"] + ball["radius"] > WIDTH:
            ball["vx"] = -ball["vx"]
        if ball["y"] - ball["radius"] < 0 or ball["y"] + ball["radius"] > HEIGHT:
            ball["vy"] = -ball["vy"]

        # Check for collisions with other balls
        for j, other in enumerate(balls):
            if i != j:
                dx = ball["x"] - other["x"]
                dy = ball["y"] - other["y"]
                distance = math.sqrt(dx**2 + dy**2)
                min_distance = ball["radius"] + other["radius"]

                if distance < min_distance:
                    # Separate the balls
                    overlap = min_distance - distance
                    nx, ny = dx / distance, dy / distance
                    ball["x"] += nx * (overlap / 2)
                    ball["y"] += ny * (overlap / 2)
                    other["x"] -= nx * (overlap / 2)
                    other["y"] -= ny * (overlap / 2)

                    # Handle elastic collision
                    relative_vx = ball["vx"] - other["vx"]
                    relative_vy = ball["vy"] - other["vy"]
                    dot_product = relative_vx * nx + relative_vy * ny
                    if dot_product > 0:  # Ensure balls are moving apart
                        continue

                    impulse = 2 * dot_product / 2
                    ball["vx"] -= impulse * nx
                    ball["vy"] -= impulse * ny
                    other["vx"] += impulse * nx
                    other["vy"] += impulse * ny
